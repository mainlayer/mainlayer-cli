    1. Create src/cli/index.ts:
       ```typescript
       #!/usr/bin/env node
       import { Command } from 'commander';
       import { authCommand } from './auth.js';
       import { walletCommand } from './wallet.js';
       import { configCommand } from './config.js';
       import { apiClient } from '../services/api-client.js';
       import { printError } from '../utils/output.js';
       import { AppError } from '../utils/errors.js';

       const program = new Command('mainlayer')
         .version('0.1.0', '-v, --version')
         .description('Mainlayer CLI — AI-native payment infrastructure')
         .option('--json', 'Output machine-readable JSON')
         .option('--api-key <key>', 'API key override (also: MAINLAYER_API_KEY env)', process.env.MAINLAYER_API_KEY)
         .addCommand(authCommand())
         .addCommand(walletCommand())
         .addCommand(configCommand());

       // Global hook: propagate --json and --api-key to subcommands
       program.hook('preAction', (thisCommand) => {
         const opts = program.opts();
         // If --api-key provided at top level, set it on ApiClient
         if (opts.apiKey) {
           apiClient.setApiKeyOverride(opts.apiKey);
         }
       });

       // Global error handler
       program.exitOverride(); // prevent Commander from calling process.exit directly
       (async () => {
         try {
           await program.parseAsync(process.argv);
         } catch (err) {
           if (err instanceof AppError) {
             printError(err.message);
             process.exitCode = err.exitCode;
           } else if (err instanceof Error && err.constructor.name === 'CommanderError') {
             // Commander handles --help, --version, invalid usage
             // exitCode already set by Commander
           } else {
             printError(String(err));
             process.exitCode = 1;
           }
         }
       })();
       ```
